CREATE SCHEMA `muj_unity_fest-2025` ;
create table event_details (event_id integer not null auto_increment, event_name varchar(255), first_name varchar(255), fourth_name varchar(255), second_name varchar(255), team_name varchar(255), third_name varchar(255), primary key (event_id)) engine=InnoDB
create table payment_order (order_id integer not null auto_increment, amount float(53) not null, course varchar(255) not null, email varchar(255) not null, name varchar(255) not null, order_status varchar(255) not null, phone_number varchar(255) not null, razor_order_id varchar(255) not null, primary key (order_id)) engine=InnoDB
